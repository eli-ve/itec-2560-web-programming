This segment converts the lyrics to lowercase and removes any punctuation. 

song_lyrics = song_lyrics.lower()
to_remove = ['\'', ',', ':', '!', '?']
for symbol_char in to_remove:
     song_lyrics = song_lyrics.replace(symbol_char, '')
